`std::visit`

```c++
#include <variant>
struct Visitor {
    void operator()(int i) {cout<<"Int: "<<i<< endl;}
    void operator()(const std::string& s){cout<<"String: "<<s<<endl;}
}
std::visit(Visitor{}, v);

```
